\documentclass[11pt, openany]{book}

% === CONFIGURABLE BOOK METADATA ===
% Override these in your project's preamble:
%   \renewcommand{\booktitle}{My Novel}
%   \renewcommand{\booksubtitle}{A Subtitle}
%   \renewcommand{\bookauthor}{Author Name}
\newcommand{\booktitle}{Untitled Novel}
\newcommand{\booksubtitle}{A Novel}
\newcommand{\bookauthor}{Unknown Author}

% === GEOMETRY: Trade paperback (5.5 x 8.5 inches) ===
\usepackage[
  paperwidth=5.5in,
  paperheight=8.5in,
  inner=0.85in,
  outer=0.65in,
  top=0.75in,
  bottom=0.85in,
  headheight=14pt
]{geometry}

% === FONTS ===
\usepackage{fontspec}
\setmainfont{EB Garamond}[
  Ligatures=TeX,
  Numbers=OldStyle
]

% === TYPOGRAPHY ===
\usepackage{microtype}
\usepackage{setspace}
\setstretch{1.12}
\usepackage{parskip}
\setlength{\parindent}{1.5em}
\setlength{\parskip}{0pt}

% === GRAPHICS ===
\usepackage{graphicx}

% === DROP CAPS ===
\usepackage{lettrine}
\setcounter{DefaultLines}{2}

% === HEADERS AND FOOTERS ===
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\fancyhead[LE]{\small\textsc{\booktitle}}
\fancyhead[RO]{\small\textit{\leftmark}}
\fancyfoot[C]{\thepage}
\renewcommand{\headrulewidth}{0pt}

% Empty pages (chapter openers)
\fancypagestyle{plain}{%
  \fancyhf{}
  \renewcommand{\headrulewidth}{0pt}
}

% === ORNAMENT (decorative break) ===
\newcommand{\chapterornament}{%
  \begin{center}
  \vspace{0.3in}
  \rule{2in}{0.4pt}\\[0.1in]
  \textsc{\large $\diamond$}\\[0.1in]
  \rule{2in}{0.4pt}
  \vspace{0.3in}
  \end{center}
}

% === SCENE BREAK ===
\newcommand{\scenebreak}{%
  \begin{center}
  \vspace{0.2in}
  * \quad * \quad *
  \vspace{0.2in}
  \end{center}
}

% === CHAPTER NUMBERING ===
\usepackage{titlesec}
\renewcommand{\thechapter}{\Roman{chapter}}
\titleformat{\chapter}[display]
  {}
  {\chapterornament}
  {0pt}
  {\Huge\textsc}
\titlespacing*{\chapter}{0pt}{0pt}{0pt}

% Lowercase chapter name in header
\renewcommand{\chaptermark}[1]{\markboth{#1}{}}

% === TITLE PAGE ===
\newcommand{\makenoveltitle}{%
  \thispagestyle{empty}
  \begin{center}
  \vspace*{2in}
  {\Huge\textsc{\booktitle}}\\[0.4in]
  {\large\textit{\booksubtitle}}\\[0.5in]
  {\small------\quad$\diamond$\quad------}\\[0.5in]
  {\large\textit{A Novel}}\\[1in]
  {\Large\textsc{\bookauthor}}\\[1.5in]
  \end{center}
  \clearpage
}

% === EPIGRAPH (optional) ===
\newcommand{\makeepigraph}[1]{%
  \thispagestyle{empty}
  \vspace*{2.5in}
  \begin{center}
  \begin{minipage}{3in}
  \begin{center}
  \itshape
  #1
  \end{center}
  \end{minipage}
  \end{center}
  \clearpage
}

% === COPYRIGHT PAGE ===
\newcommand{\makecopyright}[1]{%
  \thispagestyle{empty}
  \vspace*{\fill}
  \begin{center}
  \small
  #1
  \end{center}
  \vspace*{\fill}
  \clearpage
}
